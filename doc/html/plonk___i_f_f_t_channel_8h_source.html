00115         const int outputBufferLength = this->getOutputBuffer().length();
00116         
00117         plonk_assert (outputBufferLength == realBuffer.length());
00118

00123 
00124         // pack

00127         for (i = 0; i <= halfLength; ++i)
00128             outputSamples[i] = realInputSamples[i];

00133         // transform
