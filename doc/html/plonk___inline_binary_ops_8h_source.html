266 template<class Type> PLONK_INLINE_HIGH Type decayFeedback (Type const& delayTime, Type const& decayTime) throw()
267 {
268  const Type zero (Math<Type>::get0());
269  const Type log001 (Math<Type>::getLog0_001());
270 
271  if (delayTime > zero) return Type (exp (log001 * delayTime / decayTime));
272  else if (delayTime < zero) return -Type (exp (log001 * delayTime / -decayTime));
273  else return zero;
