121 static PLANK_INLINE_LOW PlankB pl_ThreadSpinLock_TryLock (PlankThreadSpinLockRef p)
122 {
123  PlankThreadID threadID = pl_ThreadCurrentID();
124 
125  if (pl_AtomicL_CompareAndSwap (&p->flag, PLANK_THREADSPINLOCK_UNLOCKED, threadID))
126  return PLANK_TRUE;
127  else
128  return pl_AtomicL_CompareAndSwap (&p->flag, threadID, threadID); // stay locked on same thread
129 }
