139 template<class NumericalType>
140 int BinaryFileInternal::read (NumericalArray<NumericalType>& array) throw()
141 {
142  int bytesRead;
143  plonk_assert (array.length() > 0);
144 
145  ResultCode result = pl_File_Read (getPeerRef(),
146  reinterpret_cast<void*> (array.getArray()),
147  sizeof (NumericalType) * array.length()
148  &bytesRead);
149 
150  plonk_assert (result == PlankResult_OK || result == PlankResult_FileEOF);
151 
152  if (sizeof (NumericalType) > 1 && !isNativeEndian())
153  Endian::swap (array);
154 
155 #ifndef PLONK_DEBUG
156  (void)result;
157 #endif
158 
159  return bytesRead;
160 }

162 template<class NumericalType>
163 void BinaryFileInternal::write (NumericalArray<NumericalType> const& array) throw()
164 {
165  const int length = array.length();
166 
167  if (length > 0)
168  {
169  if (isNativeEndian() || (sizeof(NumericalType) == 1))
170  {
171  ResultCode result = pl_File_Write (getPeerRef(),
172  reinterpret_cast<const void*> (array.getArray()),
173  sizeof (NumericalType) * length);
174 
175  plonk_assert (result == PlankResult_OK);
176 
177 #ifndef PLONK_DEBUG
178  (void)result;
179 #endif
180  }
181  else
182  {
183  const NumericalType* rawArray = array.getArray();
184 
185  for (int i = 0; i < length; ++i)
186  write (rawArray[i]);
187  }
188  }
189 }
