118 class UnaryOpChannelInternal

168  void initChannel (const int channel) throw()
169  {
170  const UnitType& input = this->getInputAsUnit (IOKey::Generic);
171  const SampleType sourceValue = input.getValue (channel);
172 
173  this->setBlockSize (BlockSize::decide (input.getBlockSize (channel),
174  this->getBlockSize()));
175  this->setSampleRate (SampleRate::decide (input.getSampleRate (channel),
176  this->getSampleRate()));
177 
178  this->setOverlap (input.getOverlap (channel));
179 
180  this->initValue (op (sourceValue));
181  }
182 
183  void process (ProcessInfo& info, const int channel) throw()
184  {
185  UnitType& operandUnit (this->getInputAsUnit (IOKey::Generic));
186  const Buffer& operandBuffer (operandUnit.process (info, channel));
187 
188  SampleType* const outputSamples = this->getOutputSamples();
189  const int outputBufferLength = this->getOutputBuffer().length();
190 
191  const SampleType* const operandSamples = operandBuffer.getArray();
192  const int operandBufferLength = operandBuffer.length();
193 
194  int i;
195 
196  if (operandBufferLength == outputBufferLength)
197  {
198  NumericalArrayUnaryOp<SampleType,op>::calc (outputSamples, operandSamples, outputBufferLength);
199  }
200  else if (operandBufferLength == 1)
201  {
202  NumericalArrayFiller<SampleType>::fill (outputSamples, operandSamples[0], outputBufferLength);
203  }
204  else
205  {
206  double operandPosition = 0.0;
207  const double operandIncrement = double (operandBufferLength) / double (outputBufferLength);
208 
209  for (i = 0; i < outputBufferLength; ++i)
210  {
211  outputSamples[i] = op (operandSamples[int (operandPosition)]);
212  operandPosition += operandIncrement;
213  }
214  }
215  }
216 };
