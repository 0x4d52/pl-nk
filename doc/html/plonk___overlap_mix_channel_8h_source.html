88  void initChannel (const int channel) throw()
89  {
90  const UnitType& input = this->getInputAsUnit (IOKey::Generic);
91  const SampleType sourceValue = input.getValue (channel);
92 
93  if ((channel % this->getNumChannels()) == 0)
94  {
95  // all need to be the same input BS and SR
96  this->setBlockSize (input.getBlockSize (0));
97  this->setSampleRate (input.getSampleRate (0));
98 
99  for (int i = 0; i < this->getNumChannels(); ++i)
100  {
101  tempBuffers.atUnchecked (i).setSize (this->getBlockSize().getValue() * 2, false);
102  tempBuffers.atUnchecked (i).zero();

111  void process (ProcessInfo& info, const int /*channel*/) throw()
112  {
113  /* Be careful optimising this with the new NumericalArray vector stuff */
114 
115  UnitType& inputUnit (this->getInputAsUnit (IOKey::Generic));
116 
117  const int numChannels = this->getNumChannels();
118  const int outputBufferLength = this->getOutputBuffer (0).length();
119  const int overlapLength = int (this->overlapMix.getValue() * outputBufferLength + 0.5);
120  int i, channel, channelBufferPos;
121 
122  if (overlapLength < outputBufferLength)
123  {
124  const TimeStamp infoTimeStamp = info.getTimeStamp();
125 
126  for (channel = 0; channel < numChannels; ++channel)
127  {
128  channelBufferPos = tempBufferPos;
129 
130  SampleType* const outputSamples = this->getOutputSamples (channel);
131  SampleType* const tempBufferSamples = this->tempBuffers.atUnchecked (channel).getArray();
132 
133  // copy remaining overlap from last time to the output
134  for (i = 0; i < outputBufferLength; ++i)
135  outputSamples[i] = tempBufferSamples[channelBufferPos++];
136 
137  tempBuffers.atUnchecked (channel).zero();
138  }
139 
140  // build the overlapping material in the temp buffer
141  tempBufferPos = tempBufferStartOffset;
142 
143  while (tempBufferPos < outputBufferLength)
144  {
145  const int tempBufferStartPos = tempBufferPos;
146 
147  info.setTimeStamp (nextInputTimeStamp);
148 
149  for (channel = 0; channel < numChannels; ++channel)
150  {
151  channelBufferPos = tempBufferPos;
152 
153  SampleType* const tempBufferSamples = this->tempBuffers.atUnchecked (channel).getArray();
154 
155  const Buffer& inputBuffer (inputUnit.process (info, channel));
156  const SampleType* const inputSamples = inputBuffer.getArray();
157  const int inputBufferLength = inputBuffer.length();
158 
159  plonk_assert (inputBufferLength == outputBufferLength);
160 
161  for (i = 0; i < inputBufferLength; ++i)
162  tempBufferSamples[channelBufferPos++] += inputSamples[i];
163 
164  channelBufferPos = tempBufferStartPos + overlapLength;
165  }
166 
167  tempBufferPos = channelBufferPos;
168  nextInputTimeStamp = inputUnit.getNextTimeStamp (0);
169  }
170 
171  // in case the overlap is not an integer divsion of the block size
172  tempBufferStartOffset = tempBufferPos - outputBufferLength;
173 
174  // accumulate the first part of the temp buffer to the output
175  // leaving the rest there for next time
176  tempBufferPos = 0;
177 
178  for (channel = 0; channel < numChannels; ++channel)
179  {
180  channelBufferPos = tempBufferPos;
181 
182  SampleType* const outputSamples = this->getOutputSamples (channel);
183  SampleType* const tempBufferSamples = this->tempBuffers.atUnchecked (channel).getArray();
184 
185  for (i = 0; i < outputBufferLength; ++i)
186  outputSamples[i] += tempBufferSamples[channelBufferPos++];
187  }
188 
189  tempBufferPos = channelBufferPos;
190 
191  info.setTimeStamp (infoTimeStamp); // reset for the parent graph
192  }
193  else
194  {
195  for (channel = 0; channel < numChannels; ++channel)
196  {
197  SampleType* const outputSamples = this->getOutputSamples (channel);
198  const Buffer& inputBuffer (inputUnit.process (info, channel));
199  const SampleType* const inputSamples = inputBuffer.getArray();
200 
201  plonk_assert (outputBufferLength == inputBuffer.length());
202 
203  Buffer::copyData (outputSamples, inputSamples, outputBufferLength);
204  }
205  }
206  }
