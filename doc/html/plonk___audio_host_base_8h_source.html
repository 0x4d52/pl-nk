172  PLONK_INLINE_LOW void process() throw()
173  {
174  int i;
175 
176 #ifdef PLONK_DEBUG
177  Threading::ID currentThreadID = Threading::getCurrentThreadID();
178  if (currentThreadID != Threading::getAudioThreadID())
179  Threading::setAudioThreadID (Threading::getCurrentThreadID());
180 #endif
181  const int numInputs = this->inputs.length();
182  const int numOutputs = this->outputs.length();
183  plonk_assert (this->busses.length() == numInputs);
184 
185  int blockRemain = preferredHostBlockSize;
186  const int graphBlockSize = BlockSize::getDefault().getValue();
187 
188  // must do more checks in case the block sizes are not compatible on this run
189 
190 
191  // push all the input samples for this hardware frame onto the busses
192  for (i = 0; i < numInputs; ++i)
193  {
194  BusType& bus = this->busses.atUnchecked (i);
195  bus.getWriteBlockSize().setValue (blockRemain);
196  bus.write (this->info.getTimeStamp(), blockRemain, this->inputs.atUnchecked (i));
197  }
198 
199  // write the hardware frame in possible smaller blocks
200  if (this->outputUnit.isNotNull())
201  {
202  while (blockRemain > 0)
203  {
204  this->outputUnit.process (this->info);
205 
206  for (i = 0; i < numOutputs; ++i)
207  {
208  const SampleType* const unitOutput = this->outputUnit.getOutputSamples (i);
209  BufferType::copyData (this->outputs.atUnchecked (i), unitOutput, graphBlockSize);
210  this->outputs.atUnchecked (i) += graphBlockSize;
211  }
212 
213  this->info.offsetTimeStamp (SampleRate::getDefault().getSampleDurationInTicks() * graphBlockSize);
214 
215  blockRemain -= graphBlockSize;
216  }
217  }
218  else if (numOutputs > 0)
219  {
220  for (i = 0; i < numOutputs; ++i)
221  BufferType::zeroData (this->outputs.atUnchecked (i), blockRemain);
222 
223  this->info.offsetTimeStamp (SampleRate::getDefault().getSampleDurationInTicks() * blockRemain);
224  }
225 
226 #if PLONK_DEBUG
227  // null the pointers to cause crash if buffers are not updated each HW block
228  this->inputs.zero();
229  this->outputs.zero();
230 #endif
231  }

273  {
274  SampleRate::getDefault().setValue (preferredHostSampleRate);
275 
276  // preferredGraphBlockSize must be less that the hardware size
277  // and divide into the hardware size without a remainder.
278  if ((preferredGraphBlockSize == 0) ||
279  (preferredHostBlockSize <= preferredGraphBlockSize) ||
280  ((preferredHostBlockSize % preferredGraphBlockSize) != 0))

292 template<class SampleType>
293 void AudioHostBase<SampleType>::setNumInputs (const int numInputs) throw()
294 {
295  plonk_assert (numInputs >= 0);
296 
297  if (numInputs != this->getNumInputs())
298  {
299  this->busses.clear();
300 
301  if (numInputs != 0)
302  {
303  inputs.setSize (numInputs, false);
304 
305  for (int i = 0; i < numInputs; ++i)
306  {
307  this->inputs.atUnchecked (i) = 0;
308  this->busses.add (BusType (i));
309  }
310  }
311  else this->inputs.clear();
312  }
313 }
