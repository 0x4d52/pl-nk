132  static PLONK_INLINE_LOW const double getFactorialInternal() throw()
133  {
134  double v = 1.0;
135 
136  for (int i = 2; i <= N; ++i)
137  v *= double (i);
138 
139  return v;
140  }
141 
142  template<class CalcType>
143  static PLONK_INLINE_LOW const double getFibonacciInternal (const int n) throw()
144  {
145  CalcType y0 = 1;
146  CalcType y1 = 1;
147 
148  for (int i = 2; i <= N; ++i)
149  {
150  const CalcType temp = y0 + y1;
151  y1 = y0;
152  y0 = temp;
153  }
154 
155  return y0;
156  }
157 
158 
159  static PLONK_INLINE_LOW const double getSumInternal() throw()
160  {
161  double v = 1.0;
162 
163  for (int i = 2; i <= N; ++i)
164  v += double (i);
165 
166  return v;
167  }
