85  void initChannel (const int channel) throw()
86  {
87  const UnitType& input = this->getInputAsUnit (IOKey::Generic);
88  const UnitType& position = this->getInputAsUnit (IOKey::Position);
89 
90  if ((channel % this->getNumChannels()) == 0)
91  {
92  // use sample rate and block size of the input unless another
93  // preference has been indicated on construction
94  this->setBlockSize (BlockSize::decide (input.getBlockSize (0),
95  this->getBlockSize()));
96  this->setSampleRate (SampleRate::decide (input.getSampleRate (0),
97  this->getSampleRate()));
98  }
99 
100  plonk_assert (input.getOverlap (0) == Math<DoubleVariable>::get1());
101  plonk_assert (position.getOverlap (0) == Math<DoubleVariable>::get1());
102 
103  const SampleType inputValue = input.getValue (0);
104  const SampleType mulAdd (0.5);
105  const SampleType rightLevel = position.getValue (0) * mulAdd + mulAdd;
106 
107  SampleType value (0);
108 
109  switch (channel)
110  {
111  case 0: {
112  const SampleType leftLevel = SampleType (1.0) - rightLevel;
113  value = inputValue * leftLevel;
114  } break;
115  case 1: {
116  value = inputValue * rightLevel;
117  } break;
118  default: {
119  plonk_assertfalse;
120  } break;
121  }
122 
123  this->initProxyValue (channel, value);
124  }

126  void process (ProcessInfo& info, const int /*channel*/) throw()
127  {
128  UnitType& inputUnit (this->getInputAsUnit (IOKey::Generic));
129  UnitType& positionUnit (this->getInputAsUnit (IOKey::Position));
130 
131  const Buffer& inputBuffer (inputUnit.process (info, 0));
132  const Buffer& positionBuffer (positionUnit.process (info, 0));
133 
134  const SampleType mulAdd (0.5);
135  const SampleType rightLevel = positionBuffer.atUnchecked (0) * mulAdd + mulAdd;
136  const SampleType leftLevel = SampleType (1.0) - rightLevel;
137 
138  const SampleType* const inputSamples = inputBuffer.getArray();
139  const int inputBufferLength = inputBuffer.length();
140 
141  SampleType* const leftOutputSamples = this->getOutputSamples (LeftOutput);
142  SampleType* const rightOutputSamples = this->getOutputSamples (RightOutput);
143  const int outputBufferLength = this->getOutputBuffer (0).length();
144 
145  int i;
146 
147  if (inputBufferLength == outputBufferLength)
148  {
149  for (i = 0; i < outputBufferLength; ++i)
150  {
151  const SampleType inputValue = inputSamples[i];
152  leftOutputSamples[i] = leftLevel * inputValue;
153  rightOutputSamples[i] = rightLevel * inputValue;
154  }
155  }
156  else
157  {
158  double inputPosition = 0.0;
159  const double inputIncrement = double (inputBufferLength) / double (outputBufferLength);
160 
161  for (i = 0; i < outputBufferLength; ++i)
162  {
163  const SampleType inputValue = inputSamples[int(inputPosition)];
164 
165  leftOutputSamples[i] = leftLevel * inputValue;
166  rightOutputSamples[i] = rightLevel * inputValue;
167 
168  inputPosition += inputIncrement;
169  }
170  }
171  }
