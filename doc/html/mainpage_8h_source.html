00088  basic synthesis working.

00090  @code
00091  #include "plonk/plonk.h"
00092  
00093  class TestPlonk
00094  {
00095  public:

00098     }

00102     {
00103         SampleRate::getDefault().setValue (sampleRate);
00104         BlockSize::getDefault().setValue (blockSize);
00105  
00106         graph = constructGraph();
00107     }
00108  
00109     // construct the graph of processing objects
00110     Unit constructGraph()
00111     {
00112         // just a 1kHz test tone at amplitude 0.25 (-12dB)
00113         return Sine::ar (1000, 0.25);
00114     }
00115  
00116     // assumes outputs is an array of pointers to output channel buffers
00117     void render (float** outputs, int numOutputs, int blockSize)
00118     {
00119         // update block size just in case it changed
00120         BlockSize::getDefault().setValue (blockSize);
00121  
00122         // process the graph 
00123         graph.process (info);
00124         
00125         // pull the data out of the root of the graph 
00126         for (i = 0; i < numOutputs; ++i)
00127         {
00128             float* const output = outputs[i];
00129             const float* const graphOutput = graph.getOutputSamples (i);
00130             Float1D::copyData (output, graphOutput, blockSize);

00132     }
