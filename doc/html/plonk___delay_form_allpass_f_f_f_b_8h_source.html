109  static PLONK_INLINE_LOW void readIgnore (Data&, DelayState&) throw() { }
110  static PLONK_INLINE_LOW void readRead (Data&, DelayState& state) throw()
111  {
112  const DurationType readPosition = DurationType (state.writePosition) - state.paramsOut[DurationInSamplesOut] + state.bufferLengthIndex;
113 // plonk_assert (readPosition >= 0 && readPosition <= state.bufferLengthIndex);
114  state.readValue = InterpType::lookup (state.bufferSamples, readPosition);
115  }
116 
117  static PLONK_INLINE_LOW void writeIgnore (Data&, DelayState&) throw() { }
118  static PLONK_INLINE_LOW void writeWrite (Data&, DelayState& state) throw()
119  {
120  plonk_assert (state.writePosition >= 0 && state.writePosition < state.bufferLength);
121  state.writeValue = state.inputValue + state.paramsOut[CoeffOut] * state.readValue;
122  state.bufferSamples[state.writePosition] = state.writeValue;
123  state.bufferSamples[state.writePosition + state.bufferLength] = state.writeValue; // for interpolation
124  state.bufferSamples[state.writePosition + state.bufferLength + state.bufferLength] = state.writeValue; // for interpolation
125 
126  }

128  static PLONK_INLINE_LOW void outputIgnore (Data&, DelayState&) throw() { }
129  static PLONK_INLINE_LOW void outputWrite (Data&, DelayState& state) throw()
130  {
131  state.outputValue = state.readValue - state.paramsOut[CoeffOut] * state.writeValue;
132  *state.outputSamples++ = state.outputValue;
133  state.writePosition++;
134  }

163  static PLONK_INLINE_LOW UnitType ar (UnitType const& input,
164  DurationUnitType const& duration,
165  CoeffUnitType const& coeff,
166  const DurationType maximumDuration,
167  UnitType const& mul,
168  UnitType const& add,
169  BlockSize const& preferredBlockSize,
170  SampleRate const& preferredSampleRate) throw()
171  {
172  const Data data = { { -1.0, -1.0 }, maximumDuration, 0 };
173 
174  const int numInputChannels = input.getNumChannels();
175  const int numDurationChannels = duration.getNumChannels();
176  const int numCoeffChannels = coeff.getNumChannels();
177  const int numChannels = plonk::max (numInputChannels, numDurationChannels, numCoeffChannels);
178 
179  UnitType mainUnit = UnitType::emptyChannels (numChannels);
180 
181  for (int i = 0; i < numChannels; ++i)
182  {
183  Inputs inputs;
184  inputs.put (IOKey::Generic, input[i]);
185  inputs.put (IOKey::Duration, duration[i]);
186  inputs.put (IOKey::Coeffs, coeff[i]);
187 
188  UnitType unit = UnitType::template proxiesFromInputs<DelayInternal> (inputs,
189  data,
190  preferredBlockSize,
191  preferredSampleRate);
192  mainUnit.put (i, unit);
193  }
194 
195  return UnitType::applyMulAdd (mainUnit, mul, add);
196  }
