172 template<class SampleType>
173 void AudioFileReaderInternal::readFrames (NumericalArray<SampleType>& data,
174  const bool applyScaling,
175  const bool deinterleave,
176  IntVariable& numLoops) throw()
177 {
178  typedef NumericalArray<SampleType> Buffer;
179 
180  this->hitEndOfFile = false;
181  this->numChannelsChanged = false;
182  this->audioFileChanged = false;
183 
184  ResultCode result = PlankResult_OK;
185 
186  const int dataLength = data.length();
187  int dataRemaining = dataLength;
188 
189  SampleType* dataArray = data.getArray();
190  void* const readBufferArray = readBuffer.getArray();
191 
192  int encoding = getEncoding();
193  int channels = getNumChannels();
194  int bytesPerSample = getBytesPerSample();
195 
196  plonk_assert ((encoding >= AudioFile::EncodingMin) && (encoding <= AudioFile::EncodingMax));
197  plonk_assert (getBitsPerSample() > 0);
198  plonk_assert (channels > 0);
199  plonk_assert (getBytesPerFrame() > 0);
200 
201  bool isPCM = encoding & AudioFile::EncodingFlagPCM;
202  bool isFloat = encoding & AudioFile::EncodingFlagFloat;
203  bool isBigEndian = encoding & AudioFile::EncodingFlagBigEndian;
204  bool isInterleaved = !(encoding & AudioFile::EncodingFlagNonIntervleaved);
205 
206  int dataIndex = 0;
207  const int numFailsAllowed = 3;
208  int numFails = 0;
209 
210  if (!numFramesPerBuffer)
211  {
212  if (!getBytesPerFrame())
213  goto exit;
214 
215  numFramesPerBuffer = readBuffer.length() / getBytesPerFrame();
216  }
217 
218  while ((dataRemaining > 0) &&
219  (result != PlankResult_FileEOF) &&
220  (result != PlankResult_AudioFileFrameFormatChanged) &&
221  (result != PlankResult_AudioFileChanged) &&
222  (numFails < numFailsAllowed))
223  {
224  AtomicLongLong newPosition (-1);
225  newPositionOnNextRead.swapWith (newPosition);
226 
227  if (newPosition.getValueUnchecked() >= 0)
228  {
229  result = pl_AudioFileReader_SetFramePosition (getPeerRef(), newPosition.getValueUnchecked());
230  plonk_assert (result == PlankResult_OK); // just continue though in release
231  }
232 
233  const int framesToRead = plonk::min (dataRemaining / channels, numFramesPerBuffer);
234 
235  if (framesToRead == 0)
236  break; // not enough data left for one frame
237 
238  int framesRead;
239  result = pl_AudioFileReader_ReadFrames (getPeerRef(), PLANK_FALSE, framesToRead, readBufferArray, &framesRead);
240  plonk_assert ((result == PlankResult_OK) ||
241  (result == PlankResult_FileEOF) ||
242  (result == PlankResult_AudioFileFrameFormatChanged) ||
243  (result == PlankResult_AudioFileChanged));
244 
245  if (framesRead > 0)
246  {
247  const int samplesRead = framesRead * channels;
248  plonk_assert (samplesRead <= dataRemaining);
249 
250  if (isPCM)
251  {
252  if (bytesPerSample == 2)
253  {
254  Short* const convertBuffer = static_cast<Short*> (readBufferArray);
255  swapEndianIfNotNative (convertBuffer, samplesRead, isBigEndian);
256  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
257  }
258  else if (bytesPerSample == 3)
259  {
260  Int24* const convertBuffer = static_cast<Int24*> (readBufferArray);
261  swapEndianIfNotNative (convertBuffer, samplesRead, isBigEndian);
262  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
263  }
264  else if (bytesPerSample == 4)
265  {
266  Int* const convertBuffer = static_cast<Int*> (readBufferArray);
267  swapEndianIfNotNative (convertBuffer, samplesRead, isBigEndian);
268  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
269  }
270  else if (bytesPerSample == 1)
271  {
272  Char* const convertBuffer = static_cast<Char*> (readBufferArray);
273  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
274  }
275  else
276  {
277  plonk_assertfalse;
278  return;
279  }
280  }
281  else if (isFloat)
282  {
283  if (bytesPerSample == 4)
284  {
285  Float* const convertBuffer = static_cast<Float*> (readBufferArray);
286  swapEndianIfNotNative (convertBuffer, samplesRead, isBigEndian);
287  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
288  }
289  else if (bytesPerSample == 8)
290  {
291  Double* const convertBuffer = static_cast<Double*> (readBufferArray);
292  swapEndianIfNotNative (convertBuffer, samplesRead, isBigEndian);
293  Buffer::convert (dataArray, convertBuffer, samplesRead, applyScaling);
294  }
295  else
296  {
297  plonk_assertfalse;
298  return;
299  }
300  }
301 
302  if (deinterleave && isInterleaved)
303  {
304  plonk_assertfalse; // haven't tested this yet...
305  SampleType* const deinterleaveBuffer = static_cast<SampleType*> (readBufferArray);
306  Buffer::deinterleave (deinterleaveBuffer, dataArray, samplesRead, channels);
307  Buffer::copyData (dataArray, deinterleaveBuffer, samplesRead);
308  }
309  else if (!deinterleave && !isInterleaved)
310  {
311  plonk_assertfalse; // haven't tested this yet...
312  SampleType* const interleaveBuffer = static_cast<SampleType*> (readBufferArray);
313  Buffer::interleave (interleaveBuffer, dataArray, samplesRead / channels, channels);
314  Buffer::copyData (dataArray, interleaveBuffer, samplesRead);
315  }
316 
317  dataArray += samplesRead;
318  dataIndex += samplesRead;
319  dataRemaining -= samplesRead;
320  }
321  else
322  {
323  numFails++;
324  }
325 
326  if ((result == PlankResult_FileEOF) && ((numLoops.getValue() == 0) || (numLoops.getValue() > 1)))
327  {
328  result = pl_AudioFileReader_ResetFramePosition (getPeerRef());
329  plonk_assert (result == PlankResult_OK);
330 
331  if (numLoops.getValue() > 1)
332  numLoops.setValue (numLoops.getValue() - 1);
333  }
334  else if (result == PlankResult_AudioFileFrameFormatChanged)
335  {
336  encoding = getEncoding();
337 
338  const int newNumChannels = getNumChannels();
339 
340  if (newNumChannels == channels)
341  {
342  result = PlankResult_AudioFileChanged;
343  }
344  else
345  {
346  channels = getNumChannels();
347  }
348 
349  bytesPerSample = getBytesPerSample();
350 
351  plonk_assert ((encoding >= AudioFile::EncodingMin) && (encoding <= AudioFile::EncodingMax));
352  plonk_assert (getBitsPerSample() > 0);
353  plonk_assert (channels > 0);
354  plonk_assert (getBytesPerFrame() > 0);
355 
356  isPCM = encoding & AudioFile::EncodingFlagPCM;
357  isFloat = encoding & AudioFile::EncodingFlagFloat;
358  isBigEndian = encoding & AudioFile::EncodingFlagBigEndian;
359  isInterleaved = !(encoding & AudioFile::EncodingFlagNonIntervleaved);
360 
361  if (!getBytesPerFrame())
362  goto exit;
363 
364  numFramesPerBuffer = readBuffer.length() / getBytesPerFrame();
365  }
366  }
367 
368 exit:
369  if (dataIndex < dataLength)
370  data.setSize (dataIndex, true);
371 
372  this->hitEndOfFile = (result == PlankResult_FileEOF);
373  this->numChannelsChanged = (result == PlankResult_AudioFileFrameFormatChanged);
374  this->audioFileChanged = (result == PlankResult_AudioFileChanged);
375 }
