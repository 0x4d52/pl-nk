117  void process (ProcessInfo& info, const int /*channel*/) throw()
118  {
119  Data& data = this->getState();

128  void processFade (ProcessInfo& info)
129  {
130  Data& data = this->getState();
131  const int numChannels = this->getNumChannels();
132 
133  const SampleType fadeIncrement = data.fadeIncrement;
134  const SampleType fadeMax = TypeUtility<SampleType>::getTypePeak();
135  const SampleType fadeMin = Math<SampleType>::get0();
136  SampleType fadeOutLevel (0), fadeInLevel;
137 
138  for (int channel = 0; channel < numChannels; ++channel)
139  {
140  fadeOutLevel = data.fadeLevel;
141  fadeInLevel = fadeMax - fadeOutLevel;
142 
143  Buffer& outputBuffer = this->getOutputBuffer (channel);
144  SampleType* const outputSamples = outputBuffer.getArray();
145  const int outputBufferLength = outputBuffer.length();
146 
147  const Buffer& fadeInSourceBuffer (currentSource.process (info, channel));
148  const SampleType* const fadeInSourceSamples = fadeInSourceBuffer.getArray();
149  const int fadeInSourceBufferLength = fadeInSourceBuffer.length();
150 
151  const Buffer& fadeOutSourceBuffer (fadeSource.process (info, channel));
152  const SampleType* const fadeOutSourceSamples = fadeOutSourceBuffer.getArray();
153  const int fadeOutSourceBufferLength = fadeOutSourceBuffer.length();
154 
155  int i;
156 
157  if ((fadeInSourceBufferLength == outputBufferLength) &&
158  (fadeInSourceBufferLength) == fadeOutSourceBufferLength)
159  {
160  for (i = 0; i < outputBufferLength; ++i)
161  {
162  outputSamples[i] = fadeInSourceSamples[i] * fadeInLevel + fadeOutSourceSamples[i] * fadeOutLevel;
163  fadeInLevel = plonk::min (fadeInLevel + fadeIncrement, fadeMax);
164  fadeOutLevel = plonk::max (fadeOutLevel - fadeIncrement, fadeMin);
165  }
166  }
167  else
168  {
169  double fadeInSourcePosition = 0.0;
170  const double fadeInSourceIncrement = double (fadeInSourceBufferLength) / double (outputBufferLength);
171  double fadeOutSourcePosition = 0.0;
172  const double fadeOutSourceIncrement = double (fadeOutSourceBufferLength) / double (outputBufferLength);
173 
174  for (i = 0; i < outputBufferLength; ++i)
175  {
176  outputSamples[i] = fadeInSourceSamples[int (fadeInSourcePosition)] * fadeInLevel +
177  fadeOutSourceSamples[int (fadeOutSourcePosition)] * fadeOutLevel;
178 
179  fadeInLevel = plonk::min (fadeInLevel + fadeIncrement, fadeMax);
180  fadeOutLevel = plonk::max (fadeOutLevel - fadeIncrement, fadeMin);
181 
182  fadeInSourcePosition += fadeInSourceIncrement;
183  fadeOutSourcePosition += fadeOutSourceIncrement;
184  }
185  }
186 
187  if (data.allowAutoDelete == false)
188  info.resetShouldDelete();
189  }
190 
191  data.fadeLevel = fadeOutLevel;
192 
193  if (fadeOutLevel <= fadeMin)
194  {
195  this->update (Text::getMessagePatchEnd(), fadeSource);
196  fadeSource = UnitType::getNull();
197  }
198  }
199 
200  void processCopy (ProcessInfo& info)
201  {
202  Data& data = this->getState();
203  const int numChannels = this->getNumChannels();
204 
205  for (int channel = 0; channel < numChannels; ++channel)
206  {
207  Buffer& outputBuffer = this->getOutputBuffer (channel);
208  SampleType* const outputSamples = outputBuffer.getArray();
209  const int outputBufferLength = outputBuffer.length();
210 
211  const Buffer& sourceBuffer (currentSource.process (info, channel));
212  const SampleType* const sourceSamples = sourceBuffer.getArray();
213  const int sourceBufferLength = sourceBuffer.length();
214 
215  int i;
216 
217  if (sourceBufferLength == outputBufferLength)
218  {
219  Buffer::copyData (outputSamples, sourceSamples, outputBufferLength);
220  }
221  else if (sourceBufferLength == 1)
222  {
223  NumericalArrayFiller<SampleType>::fill (outputSamples, sourceSamples[0], outputBufferLength);
224  }
225  else
226  {
227  double sourcePosition = 0.0;
228  const double sourceIncrement = double (sourceBufferLength) / double (outputBufferLength);
229 
230  for (i = 0; i < outputBufferLength; ++i)
231  {
232  outputSamples[i] = sourceSamples[int (sourcePosition)];
233  sourcePosition += sourceIncrement;
234  }
235  }
236 
237  if (data.allowAutoDelete == false)
238  info.resetShouldDelete();
239  }
240  }

258  PLONK_INLINE_LOW void updateSources (ProcessInfo& info) throw()
259  {
260  UnitVariableType& var = ChannelInternalCore::getInputAs<UnitVariableType> (IOKey::UnitVariable);
261 
262  if (var.getValue() != getDummy())
263  {
264  fadeSource = currentSource;
265  currentSource = getDummy();
266  var.swapValues (currentSource);
267 
268  this->update (Text::getMessagePatchStart(), currentSource);
269 
270  Data& data = this->getState();
271 
272  DurationUnitType& durationUnit = ChannelInternalCore::getInputAs<DurationUnitType> (IOKey::Duration);
273  plonk_assert (durationUnit.getNumChannels() == 1);
274  const DurationBufferType& durationBuffer (durationUnit.process (info, 0));
275  const DurationType duration = durationBuffer.atUnchecked (0);
276 
277  if (duration > Math<DurationType>::get0())
278  {
279  const int fadeSamplesRemaining = int (duration * data.base.sampleRate + 0.5);
280  data.fadeIncrement = TypeUtility<SampleType>::getTypePeak() / SampleType (fadeSamplesRemaining);
281  data.fadeLevel = TypeUtility<SampleType>::getTypePeak();
282  }
283  }
284  }

348  static UnitType ar (UnitVariableType const& initialSource,
349  const bool allowAutoDelete = true,
350  const int preferredNumChannels = 0,

352  BlockSize const& preferredBlockSize = BlockSize::getDefault(),
353  SampleRate const& preferredSampleRate = SampleRate::getDefault()) throw()
354  {
355  Inputs inputs;
356  inputs.put (IOKey::Unknown, UnitType::getNull()); // ensure getNull() is called off the audio thread for the first time

361 
362  return UnitType::template proxiesFromInputs<PatchChannelInternalType> (inputs,

364  preferredBlockSize,
365  preferredSampleRate);
