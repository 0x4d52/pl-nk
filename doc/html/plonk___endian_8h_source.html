44 template<UnsignedLong length>
45 class EndianUtility
46 {
47 public:
48  static PLONK_INLINE_MID void swap (void* bits) throw()
49  {
50  unsigned char* data = reinterpret_cast<unsigned char*> (bits);
51  const unsigned int numSwap = length / 2;
52  const unsigned int end = length - 1;
53 
54  for (int i = 0; i < numSwap; ++i)
55  {
56  const unsigned char temp = data[i];
57  data[i] = data[end - i];
58  data[end - i] = temp;
59  }
60  }
61 
62  static PLONK_INLINE_MID void swap (void* bits,
63  const unsigned int itemSize, // hmm, isn't itemSize the same as the tparam length?
64  const unsigned long numItems) throw()
65  {
66  unsigned char* data = reinterpret_cast<unsigned char*> (bits);
67 
68  for (int i = 0; i < numItems; ++i, data += itemSize)
69  swap (reinterpret_cast<void*> (data));
70  }
71 };
