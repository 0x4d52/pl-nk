00044 template<UnsignedLong length>
00045 class EndianUtility
00046 {
00047 public:
00048     static inline void swap (void* bits) throw()
00049     {
00050         unsigned char* data = reinterpret_cast<unsigned char*> (bits);
00051         const unsigned int numSwap = length / 2;
00052         const unsigned int end = length - 1;
00053                 
00054         for (int i = 0; i < numSwap; ++i)
00055         {
00056             const unsigned char temp = data[i];
00057             data[i] = data[end - i];
00058             data[end - i] = temp;
00059         }
00060     }    
00061     
00062     static inline void swap (void* bits, 
00063                              const unsigned int itemSize, // hmm, isn't itemSize the same as the tparam length?
00064                              const unsigned long numItems) throw()
00065     {
00066         unsigned char* data = reinterpret_cast<unsigned char*> (bits);
00067 
00068         for (int i = 0; i < numItems; ++i, data += itemSize)
00069             swap (reinterpret_cast<void*> (data));
00070     }
00071 };
