104  this->setBlockSize (BlockSize::decide (leftUnit.getBlockSize (channel).selectMax (rightUnit.getBlockSize (channel)),\
105  this->getBlockSize()));\

109  const DoubleVariable rightOverlap = rightUnit.getOverlap (channel);\
110  if (leftOverlap == rightOverlap) this->setOverlap (leftOverlap);\
111  else if (leftUnit.isConstant (channel)) this->setOverlap (rightOverlap);\
112  else if (rightUnit.isConstant (channel)) this->setOverlap (leftOverlap);\
