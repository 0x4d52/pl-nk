102  void process (ProcessInfo& info, const int channel) throw()
103  {
104  UnitType& inputUnit = this->getInputAsUnit (IOKey::Generic);
105  Busses& busses = this->getInputAsBusses (IOKey::Busses);
106 
107  plonk_assert (busses.length() == inputUnit.getNumChannels());
108 
109  const int outputBufferLength = this->getOutputBuffer().length();
110  SampleType* const outputSamples = this->getOutputSamples();
111 
112  TimeStamp infoTimeStamp = info.getTimeStamp();
113  const TimeStamp latestTimeNeeded = infoTimeStamp + this->getSampleDurationInTicks() * outputBufferLength;
114 
115  while (busses.atUnchecked (0).getLatestValidTime() <= latestTimeNeeded)
116  {
117  const TimeStamp& thisTimeStamp = inputUnit.wrapAt (0).getNextTimeStamp();
118  info.setTimeStamp (thisTimeStamp);
119 
120  for (int i = 0; i < busses.length(); ++i)
121  {
122  const Buffer& inputBuffer (inputUnit.process (info, i));
123  const SampleType* const inputSamples = inputBuffer.getArray();
124  const int inputBufferLength = inputBuffer.length();
125 
126  busses.atUnchecked (i).write (thisTimeStamp, inputBufferLength, inputSamples);
127  }
128  }
129 
130  info.setTimeStamp (infoTimeStamp); // ensure it is reset for parent graph
131 
132  // output one channel..
133  busses.atUnchecked (channel).read (nextValidReadTime, outputBufferLength, outputSamples);
134  }
135 
136 private:
137  TimeStamp nextValidReadTime;
138 };

167  static PLONK_INLINE_LOW UnitInfos getInfo() throw()
168  {
169  const double blockSize = (double)BlockSize::getDefault().getValue();
170 
171  return UnitInfo ("Reblock", "Re-buffer to a different block size.",
172 
173  // output
174  ChannelCount::VariableChannelCount,
175  IOKey::Generic, Measure::None, IOInfo::NoDefault, IOLimit::None, IOKey::End,
176 
177  // inputs
178  IOKey::Generic, Measure::None, IOInfo::NoDefault, IOLimit::None,
179  IOKey::BlockSize, Measure::Samples, blockSize, IOLimit::Minimum, Measure::Samples, 1.0,
180  IOKey::End);
181  }
182 
184  static UnitType ar (UnitType const& input,
185  BlockSize const& preferredBlockSize = BlockSize::getDefault()) throw()
186  {
187  plonk_assert (input.channelsHaveSameBlockSize());
188  plonk_assert (input.channelsHaveSameSampleRate());
189  plonk_assert (input.channelsHaveSameOverlap());
190 
191  const int numChannels = input.getNumChannels();
192  Busses busses (Busses::emptyWithAllocatedSize (numChannels));
193 
194  BlockSize inputBlockSize = input.getBlockSize (0);
195  BlockSize bufferSize = preferredBlockSize.max (inputBlockSize) * 2;
196  SampleRate sampleRate = input.getSampleRate (0);
197 
198  for (int i = 0; i < numChannels; ++i)
199  busses.add (Bus (bufferSize, inputBlockSize, sampleRate));
200 
201  Inputs inputs;
202  inputs.put (IOKey::Generic, input);
203  inputs.put (IOKey::Busses, busses);
204 
205  Data data = { -1.0, -1.0 };
206 
207  return UnitType::template createFromInputs<ReblockInternal> (inputs,
208  data,
209  preferredBlockSize,
210  SampleRate::noPreference());
211  }
212 };
