202 template<class SampleType>
203 class FFTChannelInternal
204 : public ChannelInternal<SampleType, ChannelInternalCore::Data>
205 {
206 public:
207  typedef ChannelInternalCore::Data Data;
208  typedef ChannelBase<SampleType> ChannelType;
209  typedef ObjectArray<ChannelType> ChannelArrayType;
210  typedef FFTChannelInternal<SampleType> FFTInternal;
211  typedef ChannelInternal<SampleType,Data> Internal;
212  typedef ChannelInternalBase<SampleType> InternalBase;
213  typedef UnitBase<SampleType> UnitType;
214  typedef InputDictionary Inputs;
215  typedef NumericalArray<SampleType> Buffer;
216  typedef FFTEngineBase<SampleType> FFTEngineType;
217 
218  FFTChannelInternal (Inputs const& inputs,
219  Data const& data,
220  BlockSize const& blockSize,
221  SampleRate const& sampleRate) throw()
222  : Internal (inputs, data, blockSize, sampleRate)
223  {
224  }
225 
226  Text getName() const throw()
227  {
228  return "FFT";
229  }

231  IntArray getInputKeys() const throw()
232  {
233  const IntArray keys (IOKey::Generic);
234  return keys;
235  }
236 
237  InternalBase* getChannel (const int index) throw()
238  {
239  const Inputs channelInputs = this->getInputs().getChannel (index);
240  return new FFTInternal (channelInputs,
241  this->getState(),
242  this->getBlockSize(),
243  this->getSampleRate());
244  }
245 
246  void initChannel (const int channel) throw()
247  {
248  const UnitType& input = this->getInputAsUnit (IOKey::Generic);
249 
250  this->setBlockSize (input.getBlockSize (channel));
251  this->setSampleRate (input.getSampleRate (channel));
252  this->setOverlap (input.getOverlap (channel));
253  this->fft = FFTEngineType (input.getBlockSize (channel));
254 
255  this->initValue (SampleType (0)); // not really applicable with an FFT output
256  }
257 
258  void process (ProcessInfo& info, const int channel) throw()
259  {
260  /* Be careful optimising this with the new NumericalArray vector stuff */
261 
262  UnitType& inputUnit (this->getInputAsUnit (IOKey::Generic));
263  const Buffer& inputBuffer (inputUnit.process (info, channel));
264  const SampleType* const inputSamples = inputBuffer.getArray();
265 
266  SampleType* const outputSamples = this->getOutputSamples();
267  const int outputBufferLength = this->getOutputBuffer().length();
268 
269  plonk_assert (outputBufferLength == inputBuffer.length());
270 
271  if (outputBufferLength != this->fft.length())
272  this->fft = FFTEngineType (outputBufferLength);
273 
274  // transform
275  this->fft.forward (outputSamples, inputSamples);
276  }
277 
278 private:
279  FFTEngineType fft;
280 };

301 class FFTUnit
302 {
303 public:
304  typedef FFTChannelInternal<SampleType> FFTInternal;
305  typedef typename FFTInternal::Data Data;
306  typedef UnitBase<SampleType> UnitType;

308 
309  static PLONK_INLINE_LOW UnitInfos getInfo() throw()
310  {
311  return UnitInfo ("FFT", "Transforms time domain to frequency domain data.",
312 
313  // output
314  ChannelCount::VariableChannelCount,
315  IOKey::FFTPacked, Measure::FFTPacked, IOInfo::NoDefault, IOLimit::None,
316  IOKey::End,
317 
318  // inputs
319  IOKey::Generic, Measure::None,
320  IOKey::End);
321  }
322 
324  static PLONK_INLINE_LOW UnitType ar (UnitType const& input) throw()
325  {
326  // re: full templating - could say that FFT/IFFT are only supported with float (and eventually double)?
327 
328  Inputs inputs;
329  inputs.put (IOKey::Generic, input);
330 
331  Data data = { -1.0, -1.0 };
332 
333  return UnitType::template createFromInputs<FFTInternal> (inputs,
334  data,
335  BlockSize::noPreference(),
336  SampleRate::noPreference());
337  }
338 };
