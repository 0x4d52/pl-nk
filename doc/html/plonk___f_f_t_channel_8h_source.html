00108

00111         
00112         // transform
00113         this->fft.forward (realOutputSamples, inputSamples);
00114

00116         
00117         // unpack, assuming samples that should be zero are already zero.. (as they should be).
00118         for (i = this->fft.halfLength() + 1, j = 1; i < outputBufferLength; ++i, ++j)

00120             imagOutputSamples[j] = realOutputSamples[i];

00122         }
00123     }
