00142     static void process (SampleType* const outputSamples,
00143                          const int outputLength,
00144                          UnitType& inputUnit, 
00145                          UnitType& coeffsUnit, 
00146                          Data& data,
00147                          ProcessInfo& info, 
00148                          const int channel) throw()
00149     {
00150         const int firstCoeff = NumCoeffs * channel;
00151         const Buffer& inputBuffer (inputUnit.process (info, channel));
00152         const SampleType* const inputSamples = inputBuffer.getArray();
00153         
00154         const Buffer& a0Buffer (coeffsUnit.process (info, firstCoeff + CoeffA0));
00155         const Buffer& a1Buffer (coeffsUnit.process (info, firstCoeff + CoeffA1));
00156         const Buffer& b1Buffer (coeffsUnit.process (info, firstCoeff + CoeffB1));
00157         
00158         const SampleType* const a0Samples = a0Buffer.getArray();
00159         const SampleType* const a1Samples = a1Buffer.getArray();
00160         const SampleType* const b1Samples = b1Buffer.getArray();

00169         int i;
00170         
00171         if (inputLength == outputLength)
00172         {
00173             if (a0Length == inputLength)
00174             {
00175                 for (i = 0; i < outputLength; ++i)
00176                     outputSamples[i] = process (inputSamples[i], 
00177                                                 a0Samples[i], a1Samples[i], b1Samples[i], 
00178                                                 y1);
00179             }
00180             else if (a0Length == 1)
00181             {
00182                 const SampleType a0 = a0Samples[0];
00183                 const SampleType a1 = a1Samples[0];
00184                 const SampleType b1 = b1Samples[0];
00185                 
00186                 for (i = 0; i < outputLength; ++i)
00187                     for (i = 0; i < outputLength; ++i)
00188                         outputSamples[i] = process (inputSamples[i], 
00189                                                     a0, a1, b1, y1);
00190             }
00191             else

00194                 const double coeffIncrement = double (a0Length) / double (outputLength);
00195                 
00196                 for (i = 0; i < outputLength; ++i)
00197                 {
00198                     const int pos (coeffPosition);
00199                     outputSamples[i] = process (inputSamples[i], 
00200                                                 a0Samples[pos], a1Samples[pos], b1Samples[pos], 
00201                                                 y1);
00202                     coeffPosition += coeffIncrement;
00203                 }
00204             }
00205         }
00206         else
00207         {
00208             const SampleType a0 = a0Samples[0];
00209             const SampleType a1 = a1Samples[0];
00210             const SampleType b1 = b1Samples[0];
00211             
00212             double inputPosition = 0.0;
00213             const double inputIncrement = double (inputLength) / double (outputLength);
00214             
00215             for (i = 0; i < outputLength; ++i)
00216             {
00217                 outputSamples[i] = process (inputSamples[int (inputPosition)], 
00218                                             a0, a1, b1, y1);
00219                 inputPosition += inputIncrement;
00220             }            
00221         }

00223         data.y1 = zap (y1);
00224     }
