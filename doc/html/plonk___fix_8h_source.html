390  PLONK_INLINE_LOW Fix log2() const throw()
391  {
392  // Note that a negative x gives a non-real result.
393  // If x == 0, the limit of log2(x) as x -> 0 = -infinity.
394  // log2(-ve) gives a complex result.
395  if (internal <= 0)
396  return Fix::getOverflow();
397 
398  // If the input is less than one, the result is -log2(1.0 / in)
399  if (isLessThan (Fix::getOne()))
400  {
401  // Note that the inverse of this would overflow.
402  // This is the exact answer for log2(1.0 / (1<<FBits))
403  if (internal == 1)
404  return Fix (Internal (-FBits));
405 
406  return -log2Internal (reciprocal());
407  }
408 
409  // If input >= 1, just proceed as normal.
410  // Note that x == fix16_one is a special case, where the answer is 0.
411  return log2Internal (*this);
412  }
