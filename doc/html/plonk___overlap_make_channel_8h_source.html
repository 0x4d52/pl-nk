00106     void process (ProcessInfo& info, const int channel) throw()
00107     {                
00108         UnitType& inputUnit (this->getInputAsUnit (IOKey::Generic));
00109                 
00110         const int outputBufferLength = this->getOutputBuffer().length();
00111         SampleType* const outputSamples = this->getOutputSamples();
00112         
00113         const int tempBufferLength = this->tempBuffer.length();
00114         SampleType* const tempBufferSamples = this->tempBuffer.getArray();
00115         
00116         const int overlapHop = int (this->getOverlap().getValue() * outputBufferLength + 0.5);
00117         int i;
00118         
00119         if (overlapHop < outputBufferLength)
00120         {
00121             const TimeStamp infoTimeStamp = info.getTimeStamp();

00123             // ensure we have enough buffered to output the buffer
00124             while ((tempBufferPos + outputBufferLength) >= tempBufferFill)
00125             {
00126                 info.setTimeStamp (nextInputTimeStamp);
00127                 const Buffer inputBuffer (inputUnit.process (info, channel));
00128                 nextInputTimeStamp = inputUnit.getNextTimeStamp (channel);

00130                 const SampleType* const inputSamples = inputBuffer.getArray();
00131                 const int inputBufferLength = inputBuffer.length();
00132                 
00133                 plonk_assert (outputBufferLength == inputBufferLength);
00134                 
00135                 for (i = 0; i < inputBufferLength; ++i)
00136                     tempBufferSamples[tempBufferFill++] = inputSamples[i];
00137                 
00138                 plonk_assert (tempBufferFill <= tempBufferLength);
00139             }
00140             
00141             // output one buffer
00142             for (i = 0; i < outputBufferLength; ++i)
00143                 outputSamples[i] = tempBufferSamples[tempBufferPos++];
00144             
00145             // adjust the position pointer
00146             tempBufferPos -= outputBufferLength;
00147             tempBufferPos += overlapHop;
00148             
00149             // if we're over 1/2 way move the 2nd half to the start 
00150             // (the 2nd half will be re-filled next time if needed)
00151             if (tempBufferPos >= outputBufferLength)
00152             {
00153                 tempBufferFill = 0;
00154                 
00155                 for (i = outputBufferLength; i < tempBufferLength; ++i)
00156                     tempBufferSamples[tempBufferFill++] = tempBufferSamples[i];
00157                 
00158                 tempBufferPos -= outputBufferLength;
00159             }

00162         }
00163         else
00164         {
00165             const Buffer inputBuffer (inputUnit.process (info, channel));            
00166             const SampleType* const inputSamples = inputBuffer.getArray();
00167             
00168             plonk_assert (outputBufferLength == inputBuffer.length());
00169             
00170             for (i = 0; i < outputBufferLength; ++i)
00171                 outputSamples[i] = inputSamples[i];
00172             
00173         }
