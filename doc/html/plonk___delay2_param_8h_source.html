102  void process (ProcessInfo& info, const int /*channel*/) throw()
103  {
104  Data& data = this->getState();
105  DelayStateArray& delayStates = this->getDelayStates();
106 
107  UnitType& inputUnit = this->getInputAsUnit (IOKey::Generic);
108  const Buffer& inputBuffer (inputUnit.process (info, 0));
109  Param1UnitType& param1Unit = ChannelInternalCore::getInputAs<Param1UnitType> (inputKeys.atUnchecked (1));
110  Param2UnitType& param2Unit = ChannelInternalCore::getInputAs<Param2UnitType> (inputKeys.atUnchecked (2));
111 
112  plonk_assert (inputBuffer.length() == this->getOutputBuffer (0).length());
113 
114  const int writePosition = process<FormType::inputRead,
115  FormType::readRead,
116  FormType::writeWrite,
117  FormType::outputWrite,
118  FormType::param1Process,
119  FormType::param2Process>
120  (this->getOutputSamples (0),
121  this->getOutputBuffer (0).length(),
122  inputBuffer.getArray(),
123  param1Unit,
124  param2Unit,
125  data, delayStates.atUnchecked (0), info, 0);
126 
127  const int numChannels = this->getNumChannels();
128 
129  for (int i = 1; i < numChannels; ++i)
130  {
131  plonk_assert (inputBuffer.length() == this->getOutputBuffer (i).length());
132 
133  process<FormType::inputIgnore,
134  FormType::readRead,
135  FormType::writeIgnore,
136  FormType::outputWrite,
137  FormType::param1Process,
138  FormType::param2Process>
139  (this->getOutputSamples (i),
140  this->getOutputBuffer (i).length(),
141  0,
142  param1Unit,
143  param2Unit,
144  data, delayStates.atUnchecked (i), info, i);
145  }
146 
147  data.writePosition = writePosition; // update the write position from the first channel write
148  }

191 
192  if ((param1BufferLength == outputBufferLength) && (param2BufferLength == outputBufferLength))
193  {
194  while (numSamplesToProcess > 0)
195  {
196  int bufferSamplesRemaining = state.bufferLength - state.writePosition;
197  int numSamplesThisTime = plonk::min (bufferSamplesRemaining, numSamplesToProcess);
198  numSamplesToProcess -= numSamplesThisTime;

273  double param2Position = 0.0;

285  param2Function (data, state, param2Samples[int (param2Position)]);
286  FormType::template tick<inputFunction, readFunction, writeFunction, outputFunction> (data, state);
287  param1Position += param1Increment;
288  param2Position += param2Increment;
289  }
290

292  state.writePosition = 0;
293  }
294  }
