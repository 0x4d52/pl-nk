93  void initChannel (const int channel) throw()
94  {
95  int i;
96 
97  const UnitType& inputUnit = this->getInputAsUnit (IOKey::Generic);
98 
99  if ((channel % this->getNumChannels()) == 0)
100  {
101  this->setBlockSize (BlockSize::decide (inputUnit.getBlockSize (channel),
102  this->getBlockSize()));
103  this->setSampleRate (SampleRate::decide (inputUnit.getSampleRate (channel),
104  this->getSampleRate()));
105 
106  this->setOverlap (inputUnit.getOverlap (channel));
107 
108  Data& data = this->getState();
109 
110  const int bufferLength (data.maximumDuration * data.base.sampleRate + 0.5);
111 
112  for (i = 0; i < FormType::getNumCircularBuffers(); ++i)
113  circularBuffers.put (i, Buffer::newClear (bufferLength * 3));
114 
115  for (i = 0; i < this->getNumChannels(); ++i)
116  {
117  this->initProxyValue (i, SampleType (0));
118 
119  DelayState& delayState = delayStates.atUnchecked (i);
120  Buffer& circularBuffer = circularBuffers.wrapAt (i);
121 
122  Memory::zero (delayState);
123 
124  DelayStateBase& delayStateBase (reinterpret_cast<DelayStateBase&>(delayState));
125  delayStateBase.bufferSamples = circularBuffer.getArray();
126  delayStateBase.bufferLength = bufferLength,
127  delayStateBase.bufferLengthIndex = DurationType (bufferLength),
128  delayStateBase.buffer0 = DurationType (0);
129  }
130  }
131  }
132 
133  PLONK_INLINE_LOW BufferArray& getCircularBuffers() { return circularBuffers; }
134  PLONK_INLINE_LOW DelayStateArray& getDelayStates() { return delayStates; }
