93  void initChannel (const int channel) throw()
94  {
95  if ((channel % this->getNumChannels()) == 0)
96  {
97  // should look at the initial array in case no preference was given really...
98  this->setBlockSize (BlockSize::decide (BlockSize::getDefault(),

107  void process (ProcessInfo& info, const int /*channel*/) throw()
108  {
109  const Data& data = this->getState();
110 
111  const int numChannels = this->getNumChannels();
112  int i, channel;
113 
114  for (channel = 0; channel < numChannels; ++channel)
115  {
116  const Buffer& inputBuffer (currentUnit.process (info, channel));
117  const SampleType* const inputSamples = inputBuffer.getArray();
118  const int inputBufferLength = inputBuffer.length();
119 
120  Buffer& outputBuffer = this->getOutputBuffer (channel);
121  SampleType* const outputSamples = outputBuffer.getArray();
122  const int outputBufferLength = outputBuffer.length();
123 
124  if (inputBufferLength == outputBufferLength)
125  {
126  Buffer::copyData (outputSamples, inputSamples, outputBufferLength);
127  }
128  else if (inputBufferLength == 1)
129  {
130  NumericalArrayFiller<SampleType>::fill (outputSamples, inputSamples[0], outputBufferLength);
131  }
132  else
133  {
134  double inputPosition = 0.0;
135  const double inputIncrement = double (inputBufferLength) / double (outputBufferLength);
136 
137  for (i = 0; i < outputBufferLength; ++i)
138  {
139  outputSamples[i] += inputSamples[int (inputPosition)];
140  inputPosition += inputIncrement;
141  }
142  }
143 
144  if (info.getShouldDelete())
145  {
146  QueueType& queue = this->getInputAsUnitQueue (IOKey::UnitQueue);
147  currentUnit = queue.pop();
148  info.resetShouldDelete();
149  }
150  }
151  }
152 
153 private:
154  UnitType currentUnit;
155 };
