122  void process (ProcessInfo& info, const int channel) throw()
123  {
124  int i;
125 
126  SampleType* const outputSamples = this->getOutputSamples();
127  const int outputBufferLength = this->getOutputBuffer().length();
128 
129  Busses& busses (this->getInputAsBusses (IOKey::Busses));
130  Bus& bus = busses.wrapAt (channel);
131 
132  plonk_assert (bus.getSampleRate() == this->getSampleRate());
133 
134  UnitType& inputUnit (this->getInputAsUnit (IOKey::Generic));
135 
136  const Buffer& inputBuffer (inputUnit.process (info, channel));
137  const SampleType* const inputSamples = inputBuffer.getArray();
138  const int inputBufferLength = inputBuffer.length();
139 
140  if (inputBufferLength == outputBufferLength)
141  {
142  Buffer::copyData (outputSamples, inputSamples, outputBufferLength);
143  }
144  else
145  {
146  double inputPosition = 0.0;
147  double inputIncrement = double (inputBufferLength) / double (outputBufferLength);
148 
149  for (i = 0; i < outputBufferLength; ++i)
150  {
151  outputSamples[i] = inputSamples[int (inputPosition)];
152  inputPosition += inputIncrement;
153  }
154  }
155 
156  bus.write (info.getTimeStamp(), outputBufferLength, outputSamples);
157  }
