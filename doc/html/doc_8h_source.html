777 /* Creating a new iOS Project
778 
779  - In Xcode -> Preferences -> Source trees add a PL_NK_SOURCE tree to point to the pl-nk /src folder
780  - Create a project using any of the iOS templates in Xcode.
781  - change all .m files to .mm (Objective-C++)
782  - add the /src folder to the project/target (selecting don't copy files)
783  - change this to be "relative to PL_NK_SOURCE" in the right hand side-bar
784  - add $(PL_NK_SOURCE) to the header search paths in the project
785  - add PLONK_AUDIOHOST_IOS=1 to the preprocessor macros (in all configurations, debug, release etc)
786 
787  Then you need an AudioHost class e.g.,:
788 
789  @interface AudioHost : PLAudioHost<PLAudioHostDelegate>
790  @end
791 
792  @implementation AudioHost
793 
794  - (id)init
795  {
796  if (self = [super init])
797  {
798  self.delegate = self; // set which class contains the constructGraph method
799  }
800 
801  return self;
802  }
803 
804  - (Unit)constructGraphFloat:(PLAudioHost*)host
805  {
806  return Sine::ar (1000, 0.1);
807  }
808 
809  @end
810 
811  Finally you need to make one of these in the AppDelegate (or the main ViewController for single view apps would be OK):
812 
813  e.g.,:
814 
815  @interface AppDelegate : UIResponder <UIApplicationDelegate>
816 
817  @property (strong, nonatomic) UIWindow *window;
818  @property (strong, nonatomic) AudioHost *host;
819 
820  @end
821 
822  - (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions
823  {
824  _host = [[AudioHost alloc] init];
825  [_host startHost];
826 
827  return YES;
828  }
