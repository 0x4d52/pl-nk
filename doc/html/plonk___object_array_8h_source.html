550  ObjectArray (ObjectArray<ObjectType> const& array0,
551  ObjectArray<ObjectType> const& array1) throw()
552  : Base (new Internal (0, false))
553  {
554  const bool bothNullTerminated = array0.isNullTerminated() && array1.isNullTerminated();
555  const bool eitherNullTerminated = array0.isNullTerminated() || array1.isNullTerminated();
556  const int summedSizes = array0.size() + array1.size();
557  const int newSize = bothNullTerminated ? summedSizes - 1 : summedSizes;
558 
559  if (newSize > 0)
560  {
561  this->setInternal (new Internal (newSize, eitherNullTerminated));
562 
563  int newIndex = 0;
564 
565  ObjectType *array = this->getArray();
566 
567  if (array != 0)
568  {
569  const ObjectType *arrayPtr0 = array0.getArray();
570  const ObjectType *arrayPtr1 = array1.getArray();
571 
572  int i;
573  for (i = 0; i < array0.length(); ++i)
574  {
575  array[newIndex++] = arrayPtr0[i];
576  }
577 
578  for (i = 0; i < array1.length(); ++i)
579  {
580  array[newIndex++] = arrayPtr1[i];
581  }
582 
583  if (eitherNullTerminated)
584  array[newIndex] = getNullObject();
585  }
586  }
587  }
