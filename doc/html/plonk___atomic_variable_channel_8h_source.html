158  static UnitType ar (AtomicVariableType const& atomicVariable = SampleType (0),
159  BlockSize const& preferredBlockSize = BlockSize::getDefault()) throw()
160  {
161  Inputs inputs;
162  inputs.put (IOKey::AtomicVariable, atomicVariable);
163 
164  Data data = { -1.0, -1.0 };
165 
166  const SampleRate preferredSampleRate = (preferredBlockSize == BlockSize::getDefault()) ?
167  SampleRate::getDefault() :
168  (SampleRate::getDefault() / BlockSize::getDefault()) * preferredBlockSize;
169 
170  return UnitType::template createFromInputs<AtomicVariableChannelInternalType> (inputs,
171  data,
172  preferredBlockSize,
173  preferredSampleRate);
174  }
