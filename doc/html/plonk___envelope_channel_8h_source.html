116  void setTargetPoint (const int index, const int samplesRemaining) throw()
117  {
118  Data& data = this->getState();
119 
120  if ((index != -1) && (index == data.targetPointIndex)) // ?? ugh can't be right
121  {
122  // sustain
123  data.shapeState.grow.u.norm = 0;
124  data.shapeState.stepsToTarget = samplesRemaining;
125  }
126  else
127  {
128  const BreakpointsType& breakpoints = this->getInputAsBreakpoints (IOKey::Breakpoints);
129 
130  if (index < 0)
131  {
132  // start
133  data.targetPointIndex = -1;
134  data.shapeState.shapeType = Shape::Linear;
135  data.shapeState.curve = 0.f;
136  data.shapeState.grow.u.norm = SampleType (0);
137  data.shapeState.stepsToTarget = samplesRemaining;
138  }
139  else if (index >= breakpoints.getNumBreakpoints())
140  {
141  // finished...
142  data.targetPointIndex = breakpoints.getNumBreakpoints();
143  Shape::sustain (data.shapeState);
144 
145  data.done = true;
146 
147  this->update (Text::getMessageDone(), Dynamic::getNull());
148  }
149  else
150  {
151  // configure next transition
152  data.targetPointIndex = index;
153  const BreakpointType& targetBreakpoint = breakpoints.atUnchecked (index);
154 
155  const Shape& shape = targetBreakpoint.getShape();
156  data.shapeState.shapeType = shape.getType();
157  data.shapeState.curve = shape.getCurve();
158  data.shapeState.targetLevel = targetBreakpoint.getTargetLevel();
159  data.shapeState.stepsToTarget = targetBreakpoint.getTargetTime() * data.base.sampleRate;
160  Shape::initShape (data.shapeState);
161  }
162  }
163  }

165  PLONK_INLINE_LOW void nextTargetPoint (const bool gate, const int samplesRemaining) throw()
166  {
167  Data& data = this->getState();
168  const BreakpointsType& breakpoints = this->getInputAsBreakpoints (IOKey::Breakpoints);
169 
170  const int next = (data.targetPointIndex < 0) ? 0 : breakpoints.atUnchecked (data.targetPointIndex).getNext (gate);
171 
172  if ((next == BreakpointType::This) || (next == data.targetPointIndex))
173  {
174  this->setTargetPoint (data.targetPointIndex, samplesRemaining);
175  }
176  else
177  {
178  if (next == BreakpointType::Next)
179  this->setTargetPoint (data.targetPointIndex + 1, samplesRemaining);
180  else
181  this->setTargetPoint (next, samplesRemaining);
182 
183  if (data.shapeState.stepsToTarget <= 0)
184  {
185  data.shapeState.currentLevel = data.shapeState.targetLevel;
186  this->nextTargetPoint (gate, samplesRemaining);
187  }
188  }

191  void process (ProcessInfo& info, const int /*channel*/) throw()
192  {
193  Data& data = this->getState();
194 
195  SampleType* outputSamples = this->getOutputSamples();
196  const int outputBufferLength = this->getOutputBuffer().length();
197 
198  if (data.done)
199  {
200  NumericalArrayFiller<SampleType>::fill (outputSamples, data.shapeState.currentLevel, outputBufferLength);
201  }
202  else
203  {
204  UnitType& gate (this->getInputAsUnit (IOKey::Gate));
205  const Buffer& gateBuffer (gate.process (info, 0));
206  const SampleType* const gateSamples = gateBuffer.getArray();
207  const int gateBufferLength = gateBuffer.length();
208 
209  bool currGate;
210 
211  if (gateBufferLength == 1) // most efficient
212  {
213  int samplesRemaining = outputBufferLength;
214  currGate = gateSamples[0] >= SampleType (0.5);
215 
216  if (currGate != data.prevGate)
217  this->nextTargetPoint (currGate, samplesRemaining);
218 
219  while (samplesRemaining > 0)
220  {
221  const int samplesThisTime = int (plonk::min (LongLong (samplesRemaining), data.shapeState.stepsToTarget));
222 
223  if (samplesThisTime > 0)
224  {
225  Shape::processShape (data.shapeState, outputSamples, samplesThisTime);
226 
227  outputSamples += samplesThisTime;
228  samplesRemaining -= samplesThisTime;
229  }
230 
231  if (data.shapeState.stepsToTarget == 0)
232  this->nextTargetPoint (currGate, samplesRemaining);
233  }
234 
235  data.prevGate = currGate;
236  }
237  else if (gateBufferLength == outputBufferLength)
238  {
239  for (int i = 0; i < outputBufferLength; ++i)
240  {
241  currGate = gateSamples[i] >= SampleType (0.5);
242 
243  if (currGate != data.prevGate)
244  this->nextTargetPoint (currGate, 1);
245 
246  Shape::processShape (data.shapeState, outputSamples++, 1);
247 
248  if (data.shapeState.stepsToTarget == 0)
249  this->nextTargetPoint (currGate, 1);
250 
251  data.prevGate = currGate;
252  }
253  }
254  else
255  {
256  double gatePosition = 0.0;
257  const double gateIncrement = double (gateBufferLength) / double (outputBufferLength);
258 
259  for (int i = 0; i < outputBufferLength; ++i)
260  {
261  currGate = gateSamples[int (gatePosition)] >= SampleType (0.5);
262 
263  if (currGate != data.prevGate)
264  this->nextTargetPoint (currGate, 1);
265 
266  Shape::processShape (data.shapeState, outputSamples++, 1);
267 
268  if (data.shapeState.stepsToTarget == 0)
269  this->nextTargetPoint (currGate, 1);
270 
271  data.prevGate = currGate;
272  gatePosition += gateIncrement;
273  }
274  }
275 
276  if (data.done && data.deleteWhenDone)
277  info.setShouldDelete();
278  }
279  }
