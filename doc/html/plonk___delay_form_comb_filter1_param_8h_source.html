00201                                DurationUnitType const& duration,
00202                                FeedbackUnitType const& feedback,

00211         
00212         const int numInputChannels = input.getNumChannels();
00213         const int numDurationChannels = duration.getNumChannels();
00214         const int numFeedbackChannels = feedback.getNumChannels();
00215         const int numFrequencyChannels = frequency.getNumChannels();
00216         const int numChannels = plonk::max (numInputChannels, numDurationChannels, 
00217                                             numFeedbackChannels, numFrequencyChannels);
00218         
00219         UnitType mainUnit = UnitType::emptyChannels (numChannels);
00220         
00221         for (int i = 0; i < numChannels; ++i)
00222         {
00223             Inputs inputs;
00224             inputs.put (IOKey::Generic, input[i]);
00225             inputs.put (IOKey::Duration, duration[i]);
00226             inputs.put (IOKey::Feedback, feedback[i]);
00227             inputs.put (IOKey::Frequency, frequency[i]);
00228             
00229             UnitType unit = UnitType::template proxiesFromInputs<DelayInternal> (inputs,
00230                                                                                  data,
00231                                                                                  preferredBlockSize,
00232                                                                                  preferredSampleRate);
00233             mainUnit.put (i, unit);
00234         }
00235         
00236         return UnitType::applyMulAdd (mainUnit, mul, add);
00237     }
