00124         lea esi,oldValue;

00950 {
00951     char success;
00952 #if __PIC__
00953     /* If PIC is turned on, we can't use %ebx as it is reserved for the
00954      GOT pointer.  We can save and restore %ebx because GCC won't be
00955      using it for anything else (such as any of the m operands) */
00956     __asm__ __volatile__("pushl %%ebx;"   /* save ebx used for PIC GOT ptr */
00957                          "movl %6,%%ebx;" /* move new_val2 to %ebx */
00958                          "lock; cmpxchg8b %0; setz %1;"
00959                          "pop %%ebx;"     /* restore %ebx */
00960                          : "=m"(*p), "=a"(success)
00961                          : "m"(*p), "d" (oldExtra), "a" (oldPtr),
00962                          "c" (newExtra), "m" (newPtr) : "memory");
00963 #else // !__PIC__
00964     /* We can't just do the same thing in non-PIC mode, because GCC
00965      * might be using %ebx as the memory operand.  We could have ifdef'd
00966      * in a clobber, but there's no point doing the push/pop if we don't
00967      * have to. */
00968     __asm__ __volatile__("lock; cmpxchg8b %0; setz %1;"
00969                          : "=m"(*p), "=a"(success)
00970                          : "m"(*p), "d" (oldExtra), "a" (oldPtr),
00971                          "c" (newExtra), "b" (newPtr) : "memory");
