00055 static inline PlankULL pl_InterlockedCompareExchange64 (volatile PlankULL *value, 
00056                                                         PlankULL newValue, 
00057                                                         PlankULL oldValue) 
00058 {
00059     //value returned in eax::edx
00060     __asm {
00061         lea esi,oldValue;
00062         lea edi,newValue;
00063         
00064         mov eax,[esi];
00065         mov edx,4[esi];
00066         mov ebx,[edi];
00067         mov ecx,4[edi];
00068         mov esi,value;
00069         lock CMPXCHG8B [esi];                   
00070     }
00071 }
00072 #pragma warning(default:4035)
00073 #endif
