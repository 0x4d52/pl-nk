00153     static UnitType ar (UnitType const& input,
00154                         UnitType const& coeffs, 
00155                         UnitType const& mul = SampleType (1),
00156                         UnitType const& add = SampleType (0),
00157                         BlockSize const& preferredBlockSize = BlockSize::noPreference(),
00158                         SampleRate const& preferredSampleRate = SampleRate::noPreference()) throw()
00159     {             
00160         const int requiredCoeffs = FormType::NumCoeffs;
00161         const int numCoeffChannels = coeffs.getNumChannels();
00162 
00163         plonk_assert(numCoeffChannels > 0);
00164         plonk_assert((numCoeffChannels % requiredCoeffs) == 0);
00165         
00166         const int numOutputChannels = plonk::max (input.getNumChannels(), numCoeffChannels / requiredCoeffs);
00167         UnitType result (UnitType::withSize (numOutputChannels));
00168         
00169         Data data;
00170         Memory::zero (&data, sizeof (data));
00171         data.base.sampleRate = -1.0;        
00172         data.base.sampleDuration = -1.0;
00173 
00174         UnitsType groupedCoeffs = coeffs.group (requiredCoeffs);
00175         
00176         for (int i = 0; i < numOutputChannels; ++i)
00177         {
00178             Inputs inputs;
00179             inputs.put (IOKey::Signal, input[i]);
00180             inputs.put (IOKey::Coeffs, groupedCoeffs.wrapAt (i));
00181                         
00182             ChannelInternalType* internal = new FilterInternal (inputs, 
00183                                                                 data, 
00184                                                                 preferredBlockSize, 
00185                                                                 preferredSampleRate);
00186             internal->initChannel (i);
00187 
00188             result.put (i, ChannelType (internal));
00189         }
00190         
00191         return UnitType::applyMulAdd (result, mul, add);
00192     }
