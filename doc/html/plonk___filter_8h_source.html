175  static UnitType ar (UnitType const& input,
176  UnitType const& coeffs,
177  UnitType const& mul = SampleType (1),
178  UnitType const& add = SampleType (0),
179  BlockSize const& preferredBlockSize = BlockSize::noPreference(),
180  SampleRate const& preferredSampleRate = SampleRate::noPreference()) throw()
181  {
182  // re: full templating - could say that filters are only supported with float or double types?
183 
184  const int requiredCoeffs = FormType::NumCoeffs;
185  const int numCoeffChannels = coeffs.getNumChannels();
186 
187  plonk_assert (numCoeffChannels > 0);
188  plonk_assert ((numCoeffChannels % requiredCoeffs) == 0);
189 
190  const int numOutputChannels = plonk::max (input.getNumChannels(), numCoeffChannels / requiredCoeffs);
191  UnitType result (UnitType::withSize (numOutputChannels));
192 
193  Data data;
194  Memory::zero (data);
195  data.base.sampleRate = -1.0;
196  data.base.sampleDuration = -1.0;
197 
198  for (int i = 0; i < numOutputChannels; ++i)
199  {
200  Inputs inputs;
201  inputs.put (IOKey::Generic, input);
202  inputs.put (IOKey::Coeffs, coeffs);
203 
204  ChannelInternalType* internal = new FilterInternal (inputs,
205  data,
206  preferredBlockSize,
207  preferredSampleRate);
208  internal->initChannel (i);
209 
210  result.put (i, ChannelType (internal));
211  }
212 
213  return UnitType::applyMulAdd (result, mul, add);
214  }
