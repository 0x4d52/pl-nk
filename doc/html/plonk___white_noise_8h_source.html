00199 class WhiteNoiseUnit
00200 {
00201 public:    
00202     typedef WhiteNoiseChannelInternal<SampleType>   WhiteNoiseInternal;
00203     typedef typename WhiteNoiseInternal::Data       Data;
00204     typedef InputDictionary                         Inputs;

00210     static inline UnitInfos getInfo() throw()
00211     {
00212         const double blockSize = (double)BlockSize::getDefault().getValue();
00213         const double sampleRate = SampleRate::getDefault().getValue();
00214         const double peak = (double)TypeUtility<SampleType>::getTypePeak(); // will be innaccurate for LongLong
00215         
00216         return UnitInfo ("WhiteNoise", "A white noise generator.",
00217                          
00218                          // output
00219                          ChannelCount::VariableChannelCount, 
00220                          IOKey::Generic,    Measure::None,      0.0,        IOLimit::Clipped,   Measure::NormalisedBipolar, -peak, peak,
00221                          IOKey::End,
00222                          
00223                          // inputs
00224                          IOKey::Multiply,   Measure::Factor,    1.0,        IOLimit::None,
00225                          IOKey::Add,        Measure::None,      0.0,        IOLimit::None,
00226                          IOKey::BlockSize,  Measure::Samples,   blockSize,  IOLimit::Minimum,   Measure::Samples,           1.0,
00227                          IOKey::SampleRate, Measure::Hertz,     sampleRate, IOLimit::Minimum,   Measure::Hertz,             0.0,
00228                          IOKey::End);
00229     }
00230     
00232     static UnitType ar (UnitType const& mul = SampleType (1),
00233                         UnitType const& add = SampleType (0),
00234                         BlockSize const& preferredBlockSize = BlockSize::getDefault(),
00235                         SampleRate const& preferredSampleRate = SampleRate::getDefault()) throw()
00236     {                                
00237         const LongLong peak = SampleType (TypeUtility<SampleType>::getTypePeak());
00238         
00239         const int numChannels = plonk::max (mul.getNumChannels(), add.getNumChannels());
00240         UnitType result (UnitType::withSize (numChannels));
00241         
00242         Inputs inputs;
00243         
00244         for (int i = 0; i < numChannels; ++i) 
00245         {
00246             PlankRNG rng;
00247             pl_RNG_Init (&rng);
00248             pl_RNG_Seed (&rng, RNG::global().uniformInt());
00249             Data data = { { -1.0, -1.0 }, -peak, peak, rng };
00250             ChannelInternalType* internal = new WhiteNoiseInternal (inputs, 
00251                                                                     data, 
00252                                                                     preferredBlockSize, 
00253                                                                     preferredSampleRate);
00254             internal->initChannel (i);
00255             result.put (i, ChannelType (internal));
00256         }
00257         
00258         return UnitType::applyMulAdd (result, mul, add);
00259     }
00260     
00262     static UnitType kr (UnitType const& mul = SampleType (1),
00263                         UnitType const& add = SampleType (0)) throw()
00264     {
00265         return ar (mul, add, 
00266                    BlockSize::getControlRateBlockSize(), 
00267                    SampleRate::getControlRate());
00268     }        
00269 };
