273 template<class Type> PLONK_INLINE_HIGH Type q2bw (Type const& q) throw()
274 {
275  const Type q2 = q * q;
276  const Type q22 = Math<Type>::get2() * q2;
277  const Type q221 = q22 + Math<Type>::get1();
278  const Type a = q221 / q22;
279  const Type b = q221 / q2;
280  const Type b2 = b * b;
281  const Type c = plonk::sqrt (b2 * Math<Type>::get0_25() - Math<Type>::get1());
282  return plonk::log2 (a + c);
283 }

302 template<class Type>
303 PLONK_INLINE_HIGH Type zap (Type const& x) throw()
304 {
305  static const Type smallest (static_cast<Type> (1e-15));
306  static const Type biggest (static_cast<Type> (1e15));
307  const Type absx = abs (x);
308  return ((absx > smallest) && (absx < biggest)) ? x : Math<Type>::get0();
309 }
