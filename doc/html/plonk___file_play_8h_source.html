123  void process (ProcessInfo& info, const int /*channel*/) throw()
124  {
125  Data& data = this->getState();
126 
127  IntVariable& loopCount (this->template getInputAs<IntVariable> (IOKey::LoopCount));
128  int channel;
129  int offset = 0;
130  const int blockSize = this->getBlockSize().getValue();
131  int blockRemain = blockSize;
132 
133  while (offset < blockSize)
134  {
135  AudioFileReader& file = this->getInputAsAudioFileReader (IOKey::AudioFileReader);
136 
137  const int numChannels = this->getNumChannels();
138  const int fileNumChannels = file.getNumChannels();
139  int bufferSize = blockRemain * fileNumChannels;
140 
141  const LongLong numFrames = file.getNumFrames();
142  LongLong filePosition = file.getFramePosition();
143  bool willHitEOF = false;
144 
145  if (numFrames > 0)
146  {
147  LongLong framesRemaining = numFrames - filePosition;
148 
149  if (framesRemaining == 0)
150  {
151  if ((loopCount.getValue() == 0) || (loopCount.getValue() > 1))
152  {
153  file.resetFramePosition();
154  data.cueIndex = 0;
155 
156  if (loopCount.getValue() > 1)
157  loopCount.setValue (loopCount.getValue() - 1);
158  }
159 
160  filePosition = file.getFramePosition();
161  framesRemaining = numFrames - filePosition;
162  }
163 
164  if (framesRemaining <= LongLong (blockRemain))
165  {
166  bufferSize = int (plonk::min (LongLong (bufferSize), framesRemaining * fileNumChannels));
167  willHitEOF = true;
168  }
169  }
170 
171  const AudioFileMetaData metaData = file.getMetaData();
172 
173  if (metaData.isNotNull())
174  {
175  const AudioFileCuePointArray cuePoints = metaData.getCuePoints();
176  AudioFileCuePoint cue = cuePoints[data.cueIndex];
177 
178  bool renderToNextCue = false;
179 
180  if (cue.isNotNull())
181  {
182  if (cue.getFramePosition (file.getSampleRate()) == filePosition)
183  {
184  this->update (Text::getMessageCuePoint(), Text (cue.getLabel()));
185 
186  ++data.cueIndex;
187  cue = cuePoints[data.cueIndex];
188 
189  if (cue.isNotNull())
190  renderToNextCue = true;
191  }
192  else
193  {
194  renderToNextCue = true;
195  }
196 
197  if (renderToNextCue)
198  {
199  const LongLong framesToNextCue = cue.getFramePosition (file.getSampleRate()) - filePosition;
200 
201  if (framesToNextCue < LongLong (bufferSize / fileNumChannels))
202  {
203  bufferSize = int (plonk::min (LongLong (bufferSize), framesToNextCue * fileNumChannels));
204  willHitEOF = false;
205  }
206  }
207  }
208  }
209 
210  buffer.setSize (bufferSize, false);
211  file.readFrames (buffer, zero);
212  const bool changedNumChannels = file.didNumChannelsChange();
213  const bool audioFileChanged = file.didAudioFileChange();
214  const bool hitEOF = file.didHitEOF();
215  const int bufferAvailable = buffer.length();
216 
217  if ((bufferAvailable == 0) || data.done)
218  {
219  for (channel = 0; channel < numChannels; ++channel)
220  {
221  Buffer& outputBuffer = this->getOutputBuffer (channel);
222  SampleType* const outputSamples = outputBuffer.getArray() + offset;
223  const int outputBufferLength = outputBuffer.length() - offset;
224 
225  for (int i = 0; i < outputBufferLength; ++i)
226  outputSamples[i] = SampleType (0);
227  }
228 
229  offset = blockSize;
230  blockRemain = 0;
231  }
232  else if (willHitEOF || hitEOF)
233  {
234  const int bufferFramesAvailable = bufferAvailable / fileNumChannels;
235 
236  for (channel = 0; channel < numChannels; ++channel)
237  {
238  Buffer& outputBuffer = this->getOutputBuffer (channel);
239  SampleType* const outputSamples = outputBuffer.getArray() + offset;
240  const int outputBufferLength = outputBuffer.length() - offset;
241  const int outputLengthToWrite = plonk::min (bufferFramesAvailable, outputBufferLength);
242 
243  const SampleType* bufferSamples = buffer.getArray() + ((unsigned int)channel % (unsigned int)fileNumChannels);
244 
245  for (int i = 0; i < outputLengthToWrite; ++i, bufferSamples += fileNumChannels)
246  outputSamples[i] = *bufferSamples;
247  }
248 
249  if ((loopCount.getValue() == 0) || (loopCount.getValue() > 1))
250  {
251  file.resetFramePosition();
252  data.cueIndex = 0;
253 
254  if (loopCount.getValue() > 1)
255  loopCount.setValue (loopCount.getValue() - 1);
256  }
257  else if (!data.done)
258  {
259  data.done = true;
260  this->update (Text::getMessageDone(), Dynamic::getNull());
261  }
262 
263  offset += bufferFramesAvailable;
264  blockRemain -= bufferFramesAvailable;
265  }
266  else
267  {
268  const int bufferFramesAvailable = bufferAvailable / fileNumChannels;
269 
270  for (channel = 0; channel < numChannels; ++channel)
271  {
272  Buffer& outputBuffer = this->getOutputBuffer (channel);
273  SampleType* const outputSamples = outputBuffer.getArray() + offset;
274  const int outputBufferLength = outputBuffer.length() - offset;
275  const int outputLengthToWrite = plonk::min (bufferFramesAvailable, outputBufferLength);
276 
277  const SampleType* bufferSamples = buffer.getArray() + ((unsigned int)channel % (unsigned int)fileNumChannels);
278 
279  for (int i = 0; i < outputLengthToWrite; ++i, bufferSamples += fileNumChannels)
280  outputSamples[i] = *bufferSamples;
281  }
282 
283  offset += bufferFramesAvailable;
284  blockRemain -= bufferFramesAvailable;
285  }
286 
287  if (audioFileChanged)
288  this->update (Text::getMessageAudioFileChanged(), file);
289 
290  if (changedNumChannels)
291  this->update (Text::getMessageNumChannelsChanged(), IntVariable (fileNumChannels));
292  }
293 
294  if (data.done && data.deleteWhenDone)
295  info.setShouldDelete();
296  }

377  static UnitType ar (AudioFileReader const& file,
378  IntVariable const& loopCount = 0,
379  UnitType const& mul = SampleType (1),
380  UnitType const& add = SampleType (0),
381  const bool deleteWhenDone = true,
382  BlockSize const& preferredBlockSize = BlockSize::getDefault(),
383  SampleRate const& preferredSampleRate = SampleRate::noPreference()) throw()
384  {
385  if (file.isReady() && !file.isOwned())
386  {
387  Inputs inputs;
388  inputs.put (IOKey::AudioFileReader, file);
389  inputs.put (IOKey::LoopCount, loopCount);
390  inputs.put (IOKey::Multiply, mul);
391  inputs.put (IOKey::Add, add);
392 
393  Data data = { { -1.0, -1.0 }, file.getDefaultNumChannels(), 0, false, deleteWhenDone };
394 
395  return UnitType::template proxiesFromInputs<FilePlayInternal> (inputs,
396  data,
397  preferredBlockSize,
398  preferredSampleRate);
399  }
400  else return UnitType::getNull();
401  }
