00660                                         UnitBase const& mul,
00661                                         UnitBase const& add) throw()
00662     {
00663         UnitBase result = mainUnit;
00664         
00665         bool hasMul = true;
00666         bool hasAdd = true;
00667                                 
00668         if (mul.isNull() || (mul.isConstant() && (mul.getValue (0) == SampleType (1))))
00669             hasMul = false;
00670         
00671         if (add.isNull() || (add.isConstant() && (add.getValue (0) == SampleType (0))))
00672             hasAdd = false;

00676             result = MulAddUnit<SampleType>::ar (result, mul, add);
00677         else if (hasMul)

00679         else if (hasAdd)
00680             result += add;

00682             
00683         return result;
